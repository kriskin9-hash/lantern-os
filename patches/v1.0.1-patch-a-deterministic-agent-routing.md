# Patch A — Deterministic Agent Routing

**Priority:** P0
**File:** `apps/lantern-garage/lib/dream-chat.js`
**Problem:** `selectAgent()` adds `Math.random() * 3` to routing scores before sorting, making routing nondeterministic.
**Impact:** Complicates reproducibility, benchmarking, regression tests, and user trust.

## Change

Replace the random tie-breaker with a stable `index`-based tie-breaker:

```javascript
function selectAgent(message) {
  const lower = String(message || "").toLowerCase();
  const scores = AGENT_PERSONAS.map((agent, index) => {
    let score = 0;
    const keywords = {
      lantern:  ["light", "flame", "steady", "safe", "home", "glow", "protect", "lantern"],
      blinkbug: ["static", "glitch", "tv", "crt", "caterpillar", "bug", "screen", "chaotic", "unhinged", "geeked", "windows", "xp"],
      keystone: ["truth", "anchor", "memory", "story", "pattern", "integrate", "return door", "hold", "remember"],
      waterfall:["flow", "water", "heal", "gentle", "emotion", "feeling"],
      xenon:    ["space", "ship", "navigate", "map", "course", "direction"],
      founder:  ["wish", "protect", "founder", "home", "return", "safety"],
    };
    for (const kw of (keywords[agent.id] || [agent.id])) {
      if (lower.includes(kw)) score += 10;
    }
    return { agent, score, index };
  });
  scores.sort((a, b) => b.score - a.score || a.index - b.index);
  return scores[0].agent;
}
```

## Verification

Run 20 identical prompt trials and assert the same agent is selected each time:

```bash
node -e "
const { selectAgent } = require('./apps/lantern-garage/lib/dream-chat.js');
const results = [];
for (let i = 0; i < 20; i++) {
  results.push(selectAgent('I feel the light calling me').id);
}
const allSame = results.every(r => r === results[0]);
console.log(allSame ? 'PASS: deterministic' : 'FAIL: nondeterministic');
"
```
